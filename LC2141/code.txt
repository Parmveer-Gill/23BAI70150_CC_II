class Solution {
public:

    bool check(int n, vector<int>& batteries, long long mid){
        long long total = 0;
        long long required = (long long)n * mid;

        for(int b : batteries){
            total += min((long long)b, mid);
        }

        return total >= required;
    }

    long long maxRunTime(int n, vector<int>& batteries) {

        long long sum = 0;
        for(int b : batteries){
            sum += b;
        }

        long long l = 1;
        long long r = sum / n;
        long long ans = 0;

        while(l <= r){
            long long mid = l + (r - l) / 2;

            if(check(n, batteries, mid)){
                ans = mid;
                l = mid + 1;
            }
            else{
                r = mid - 1;
            }
        }

        return ans;
    }
};
